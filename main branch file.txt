package oops_concepts;

public class Parent {
	public String name;
	private int age;
	protected String location;
	public Parent(String name, int age, String location) {
		this.name = name;
		this.age = age;
		this.location = location;
	}
	public int getAge() {
		return age;
	}
	public void setAge(int age) {
		this.age = age;
	}
}
